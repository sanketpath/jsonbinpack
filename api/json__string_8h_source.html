186  // There are two-character escape sequence representations of some
187  // characters.
188  // \" represents the quotation mark character (U+0022).
189  // \\ represents the reverse solidus character (U+005C).
190  // \/ represents the solidus character (U+002F).
191  // \b represents the backspace character (U+0008).
192  // \f represents the form feed character (U+000C).
193  // \n represents the line feed character (U+000A).
194  // \r represents the carriage return character (U+000D).
195  // \t represents the character tabulation character (U+0009).
196  // See
197  // https://www.ecma-international.org/wp-content/uploads/ECMA-404_2nd_edition_december_2017.pdf
198  if (character == String<Source>::token_escape &&
199  index < string_data.size() - 1) {
200  std::string_view::const_reference next{string_data.at(index + 1)};
201  switch (next) {
202  case '\u0022':
203  case String<Source>::token_escape:
204  case '\u002F':
205  value << next;
206  index += 1;
207  continue;
208  case 'b':
209  value << '\b';
210  index += 1;
211  continue;
212  case 'f':
213  value << '\f';
214  index += 1;
215  continue;
216  case 'n':
217  value << '\n';
218  index += 1;
219  continue;
220  case 'r':
221  value << '\r';
222  index += 1;
223  continue;
224  case 't':
225  value << '\t';
226  index += 1;
227  continue;
228  case 'u':
